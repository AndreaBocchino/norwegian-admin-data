1.4 Code snippets

*-------------------------------------------------------------------------------
* 10TH GRADE GPA
*-------------------------------------------------------------------------------

* ---- globals (set these before running) ----
global raw    "/data/prosjekt"         // root of raw data
global out    "/home/user/output"      // output folder

use "${raw}/human_capital/education/tab_karakter_grunnskole_2002-2019.dta", clear

label language english

* Find year of exam
gen year = real(substr(avgdato, 1, 4))

* Keep most recent exam per individual-subject
bysort npid fagkode: gegen maxyear = max(year)
bysort npid fagkode: keep if year == maxyear

* Convert grade strings to numeric (drops letters e.g. "IBM4")
gen stp_1 = real(stp)   // subject grade (teacher-assigned)
gen skr_1 = real(skr)   // written exam grade
gen mun_1 = real(mun)   // oral exam grade

* Drop invalid grades (>6 on Norwegian scale)
replace stp_1 = . if stp_1 > 6
replace skr_1 = . if skr_1 > 6
replace mun_1 = . if mun_1 > 6

* Compute sum of available grades per subject (Fanny's algorithm + oral)
gen grades = stp_1 + skr_1 + mun_1 if !missing(stp_1) & !missing(skr_1) & !missing(mun_1)
replace grades = stp_1 + skr_1     if !missing(stp_1) & !missing(skr_1) &  missing(mun_1)
replace grades = stp_1 + mun_1     if !missing(stp_1) &  missing(skr_1) & !missing(mun_1)
replace grades = skr_1 + mun_1     if  missing(stp_1) & !missing(skr_1) & !missing(mun_1)
replace grades = stp_1             if !missing(stp_1) &  missing(skr_1) &  missing(mun_1)
replace grades = skr_1             if  missing(stp_1) & !missing(skr_1) &  missing(mun_1)
replace grades = mun_1             if  missing(stp_1) &  missing(skr_1) & !missing(mun_1)

* Aggregate to person level
bysort npid: gegen total_grades = total(grades)
bysort npid: egen nb_skr = count(skr_1)
bysort npid: egen nb_stp = count(stp_1)
bysort npid: egen nb_mun = count(mun_1)

gen gpa_10 = total_grades / (nb_skr + nb_stp + nb_mun)
drop if gpa_10 == .
drop grades total_grades nb_mun nb_skr nb_stp

* Keep one observation per individual
by npid: keep if _n == 1

save "${out}/gpa_10.dta", replace
*-------------------------------------------------------------------------------
* PANEL DATASET WITH MAX YEARS OF EDUCATION
*-------------------------------------------------------------------------------

* ---- globals (set these before running) ----
global raw    "/data/prosjekt"         // root of raw data
global out    "/home/user/output"      // output folder

use "${raw}/human_capital/education/f_demo_1974-2022.dta", clear

label language english

drop if bu_kltrinn == .

* Obtain max education years per person
bysort npid: gegen max_edu_year = max(bu_kltrinn)

order npid bu_kltrinn max_edu_year

* Keep one observation per person
bysort npid: keep if _n == 1

* Create categorical education level variable
gen max_level_edu = .
replace max_level_edu = 1 if max_edu_year <= 10
replace max_level_edu = 2 if max_edu_year == 11 | max_edu_year == 12
replace max_level_edu = 3 if max_edu_year == 13
replace max_level_edu = 4 if max_edu_year == 14 | max_edu_year == 15
replace max_level_edu = 5 if max_edu_year >= 16

label define max_level_edu_lbl        ///
    1 "Mandatory schooling or below"  ///
    2 "Some high school"              ///
    3 "Completed high school"         ///
    4 "Some college"                  ///
    5 "College and above"
label values max_level_edu max_level_edu_lbl

* Sanity check
tab max_level_edu

save "${out}/max_edu_year_level.dta", replace

1.4 Code snippets

*-------------------------------------------------------------------------------
* 10TH GRADE GPA
*-------------------------------------------------------------------------------

* ---- globals (set these before running) ----
global raw    "/data/prosjekt"         // root of raw data
global out    "/home/user/output"      // output folder

use "${raw}/human_capital/education/tab_karakter_grunnskole_2002-2019.dta", clear

label language english

* Find year of exam
gen year = real(substr(avgdato, 1, 4))

* Keep most recent exam per individual-subject
bysort npid fagkode: gegen maxyear = max(year)
bysort npid fagkode: keep if year == maxyear

* Convert grade strings to numeric (drops letters e.g. "IBM4")
gen stp_1 = real(stp)   // subject grade (teacher-assigned)
gen skr_1 = real(skr)   // written exam grade
gen mun_1 = real(mun)   // oral exam grade

* Drop invalid grades (>6 on Norwegian scale)
replace stp_1 = . if stp_1 > 6
replace skr_1 = . if skr_1 > 6
replace mun_1 = . if mun_1 > 6

* Compute sum of available grades per subject (Fanny's algorithm + oral)
gen grades = stp_1 + skr_1 + mun_1 if !missing(stp_1) & !missing(skr_1) & !missing(mun_1)
replace grades = stp_1 + skr_1     if !missing(stp_1) & !missing(skr_1) &  missing(mun_1)
replace grades = stp_1 + mun_1     if !missing(stp_1) &  missing(skr_1) & !missing(mun_1)
replace grades = skr_1 + mun_1     if  missing(stp_1) & !missing(skr_1) & !missing(mun_1)
replace grades = stp_1             if !missing(stp_1) &  missing(skr_1) &  missing(mun_1)
replace grades = skr_1             if  missing(stp_1) & !missing(skr_1) &  missing(mun_1)
replace grades = mun_1             if  missing(stp_1) &  missing(skr_1) & !missing(mun_1)

* Aggregate to person level
bysort npid: gegen total_grades = total(grades)
bysort npid: egen nb_skr = count(skr_1)
bysort npid: egen nb_stp = count(stp_1)
bysort npid: egen nb_mun = count(mun_1)

gen gpa_10 = total_grades / (nb_skr + nb_stp + nb_mun)
drop if gpa_10 == .
drop grades total_grades nb_mun nb_skr nb_stp

* Keep one observation per individual
by npid: keep if _n == 1

save "${out}/gpa_10.dta", replace
*-------------------------------------------------------------------------------
* PANEL DATASET WITH MAX YEARS OF EDUCATION
*-------------------------------------------------------------------------------

* ---- globals (set these before running) ----
global raw    "/data/prosjekt"         // root of raw data
global out    "/home/user/output"      // output folder

use "${raw}/human_capital/education/f_demo_1974-2022.dta", clear

label language english

drop if bu_kltrinn == .

* Obtain max education years per person
bysort npid: gegen max_edu_year = max(bu_kltrinn)

order npid bu_kltrinn max_edu_year

* Keep one observation per person
bysort npid: keep if _n == 1

* Create categorical education level variable
gen max_level_edu = .
replace max_level_edu = 1 if max_edu_year <= 10
replace max_level_edu = 2 if max_edu_year == 11 | max_edu_year == 12
replace max_level_edu = 3 if max_edu_year == 13
replace max_level_edu = 4 if max_edu_year == 14 | max_edu_year == 15
replace max_level_edu = 5 if max_edu_year >= 16

label define max_level_edu_lbl        ///
    1 "Mandatory schooling or below"  ///
    2 "Some high school"              ///
    3 "Completed high school"         ///
    4 "Some college"                  ///
    5 "College and above"
label values max_level_edu max_level_edu_lbl

* Sanity check
tab max_level_edu

save "${out}/max_edu_year_level.dta", replace
