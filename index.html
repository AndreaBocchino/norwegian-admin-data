1.4 Code snippets

*-------------------------------------------------------------------------------
* TRANSLATE NUS CODES TO VOCATIONAL/ACADEMIC TRACK
*-------------------------------------------------------------------------------

* ---- globals (set these before running) ----
global csv    "/data/NUS_to_vocacad"   // folder containing the downloaded CSV
global out    "/home/user/output"      // output folder

* Source: official SSB mapping at
* https://www.ssb.no/klass/klassifikasjoner/36/varianter/3442

* Import CSV downloaded from the link above
import delimited "${csv}/3442.csv", clear

* "code" is the NUS code — rename for consistency when merging with other datasets
rename code bu

* Entries 1 and 3 are track-level headers, not individual NUS codes — drop them
* 1 = Vocational (Yrkesutdanning)
* 3 = General (utdanning som ikke er ment å skulle forberede kandidatene for en spesiell yrkesgruppe)
drop if bu == 1
drop if bu == 3

* parentcode identifies the track
gen vocational = 1 if parentcode == 1   // vocational track
replace vocational = 0 if parentcode == 3   // academic/general track

save "${out}/NUS_to_vocacad_indicator.dta", replace
*-------------------------------------------------------------------------------
* PANEL DATASET WITH MAX YEARS OF EDUCATION
*-------------------------------------------------------------------------------

* ---- globals (set these before running) ----
global raw    "/data/prosjekt"         // root of raw data
global out    "/home/user/output"      // output folder

use "${raw}/human_capital/education/f_demo_1974-2022.dta", clear

label language english

drop if bu_kltrinn == .

* Obtain max education years per person
bysort npid: gegen max_edu_year = max(bu_kltrinn)

order npid bu_kltrinn max_edu_year

* Keep one observation per person
bysort npid: keep if _n == 1

* Create categorical education level variable
gen max_level_edu = .
replace max_level_edu = 1 if max_edu_year <= 10
replace max_level_edu = 2 if max_edu_year == 11 | max_edu_year == 12
replace max_level_edu = 3 if max_edu_year == 13
replace max_level_edu = 4 if max_edu_year == 14 | max_edu_year == 15
replace max_level_edu = 5 if max_edu_year >= 16

label define max_level_edu_lbl        ///
    1 "Mandatory schooling or below"  ///
    2 "Some high school"              ///
    3 "Completed high school"         ///
    4 "Some college"                  ///
    5 "College and above"
label values max_level_edu max_level_edu_lbl

* Sanity check
tab max_level_edu

save "${out}/max_edu_year_level.dta", replace
